        function kiemTra() {

            const value = document.getElementById('checkInput').value.trim();

            const result = document.getElementById('checkResult');

            if (!value) {

                result.innerHTML = '<div class="alert danger">Vui lòng nhập thông tin!</div>';

                return;

            }

            const blackList = ['scam', 'nhanqua', 'lixi', 'free', '012345'];

            let isScam = blackList.some(word => value.toLowerCase().includes(word));

            if (isScam) {

                result.innerHTML = '<div class="alert danger">⚠️ Cảnh báo: Có dấu hiệu nghi vấn!</div>';

            } else {

                result.innerHTML = '<div class="alert safe">✅ Chưa phát hiện trong danh sách đen.</div>';

            }

        }
